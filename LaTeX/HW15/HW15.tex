\section{Standard 15 - Analyzing Code: Writing Down Recurrences}
\begin{required}
For each algorithm, write down the recurrence relation for the number of times they print ``Welcome''. (In this case, this is big-$\Theta$ of the runtime - do you see why?) \textbf{Don't forget to include the base cases.}
\begin{enumerate}[label=(\alph*)]
        \subsection{Problem 1\ref{a} (2 points)}
\item \label{a}

\begin{algorithm}
\caption{Writing Recurrences 1}\label{alg:Recurrence1}
\begin{algorithmic}[1]
\Procedure{Foo}{$\text{Integer } n$}
\If{$n \leq 5$} 
\Return n
\EndIf

\noindent \\
\State $\text{Foo}(n/2)$
\State $\text{Foo}(n/4)$
\State $\text{Foo}(n/6)$ \\
    
\For{$i \gets 1; i \leq 2 * n; i \gets i+2 $}
    \State \textbf{print} \text{``Welcome"}
\EndFor
\EndProcedure
\end{algorithmic}
\end{algorithm}

\begin{proof}[Answer]
	%Need to handle i + 2
\[
T_{n} = \begin{cases} \Theta(1)& : n \leq 5, \\
	T(n/2) + T(n/4) + T(n/6) + \Theta(n)& : n > 5.
\end{cases}
\]
\end{proof}

\begin{algorithm}
\caption{Writing Recurrences 2}\label{alg:Recurrence2}
\begin{algorithmic}[1]
\Procedure{Foo2}{$\text{Integer } n$}
\If{$n \leq 7$} 
\Return
\EndIf

\noindent \\
\State $\text{Foo2}(n/14)$
\State $\text{Foo2}(n/14)$ \\
    
\For{$i \gets 1; i \leq n; i \gets i * 3 $}
    \For{$j \gets 1; j \leq n; j \gets j * 2 $}
        \State \textbf{print} \text{``Welcome"}
    \EndFor
\EndFor
\EndProcedure
\end{algorithmic}
\end{algorithm}

\begin{proof}[Answer]
	%Need to handle i * 3 and j * 2%
\[
T_{n} = \begin{cases} \Theta(1)& : n \leq 7, \\
	2T(n/14) + \Theta(\log^2(n))& : n > 7.
\end{cases}
\]
\end{proof}
