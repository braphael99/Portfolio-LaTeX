\section{Standard 11: Asymptotics I (Calculus I techniques) (4 points)}

\setcounter{required}{10}
\begin{required} 
Let $f(n)$ and $g(n)$. Determine the relationship that best applies: $f(n) \in O(g(n)), f(n) \in \Theta(g(n))$, or $f(n) \in \Omega(g(n))$. Prove your answer.
\begin{align*}
	f(n) = n^{\frac{3}{2}}\log n + n (\sin n) ~\text{ and }~ g(n) = n^{\frac{3}{2}}(\log(\log n))
\end{align*}
\end{required}

\begin{proof}[Answer]
\begin{align*}
	\lim_{n \to \infty} \left| \frac{n^{3/2}(\log(\log n))}{n^{3/2}\log n + n (\sin n)} \right| &= \left| \lim_{n \to \infty} \frac{n^{3/2}(\log(\log n))}{n^{3/2}\log n + n (\sin n)} \right| \\
	&= \left| \lim_{n \to \infty} \frac{\frac{\log(\log n)}{\log n}}{\frac{\sin n}{\sqrt{n} \log n} + 1}\right| \\
	&= \left| \frac{0}{0 + 1} \right|\\
	&= \left| 0 \right|
	&= 0.
\end{align*}

So, we have that $f(n) \in O(g(n))$.\\

\end{proof}
